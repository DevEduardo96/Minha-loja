import React from "react";
import "./css/Categorias.css";

interface CategoriasProps {
  categoriaSelecionada: string;
  setCategoriaSelecionada: (categoria: string) => void;
}

const categorias = ["todos", "ebooks", "templates", "cursos"];

const Categorias: React.FC<CategoriasProps> = ({
  categoriaSelecionada,
  setCategoriaSelecionada,
}) => {
  return (
    <nav className="categorias-menu">
      <ul>
        {categorias.map((cat) => (
          <li key={cat}>
            <button
              className={categoriaSelecionada === cat ? "ativo" : ""}
              onClick={() => setCategoriaSelecionada(cat)}
            >
              {cat.toUpperCase()}
            </button>
          </li>
        ))}
      </ul>
    </nav>
  );
};

export default Categorias;
